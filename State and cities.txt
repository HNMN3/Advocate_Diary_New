# code for creating database

state=None
f = open('State and cities.txt')
for line in f.readlines():
    line = line.strip()
    if line=="New State":
            state=None
            continue
    if state is None:
            state = State.objects.get(pk=int(line))
            continue
    c = City(name=line)
    c.state=state
    c.save()
